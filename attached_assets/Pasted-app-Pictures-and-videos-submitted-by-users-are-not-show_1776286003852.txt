In lib/app-context.tsx, replace addCasePhoto with:  async function addCasePhoto(caseId: string, photoUri: string, user?: string) {
  const now = Date.now();

  let sharedUri = photoUri;
  try {
    if (photoUri && !/^https?:\/\//i.test(photoUri) && !photoUri.startsWith("data:")) {
      const filename = photoUri.split("/").pop() || `case-photo-${Date.now()}.jpg`;
      const contentType = filename.toLowerCase().endsWith(".mp4")
        ? "video/mp4"
        : filename.toLowerCase().endsWith(".mov")
          ? "video/quicktime"
          : filename.toLowerCase().endsWith(".webp")
            ? "image/webp"
            : filename.toLowerCase().endsWith(".png")
              ? "image/png"
              : "image/jpeg";

      const host = process.env.EXPO_PUBLIC_DOMAIN;
      if (host) {
        const apiUrl = `https://${host}`;
        const response = await fetch(new URL("/api/media/share", apiUrl).toString(), {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ caseId, sourceUri: photoUri, filename, contentType }),
        });

        if (response.ok) {
          const data = await response.json();
          if (data?.url) {
            sharedUri = data.url;
          }
        }
      }
    }
  } catch (error) {
    console.log("Could not create shared case media URL:", error);
  }

  const photoEntry: ActivityEntry = {
    id: generateId(),
    type: "photo",
    timestamp: now,
    description: "Photo added to case",
    imageUri: sharedUri,
    user: user || undefined,
  };

  setCases((prevCases) => {
    const updated = prevCases.map((c) => {
      if (c.id === caseId) {
        return {
          ...c,
          updatedAt: now,
          photos: [...(c.photos || []), sharedUri],
          activityLog: [...(c.activityLog || []), photoEntry],
        };
      }
      return c;
    });
    AsyncStorage.setItem(CASES_KEY, JSON.stringify(updated));
    return updated;
  });
}

app.post("/api/media/share", async (req, res) => {
  try {
    const { caseId, sourceUri, filename, contentType } = req.body || {};
    if (!caseId || !sourceUri) {
      return res.status(400).json({ error: "caseId and sourceUri are required" });
    }

    if (/^https?:\/\//i.test(sourceUri)) {
      return res.json({ url: sourceUri });
    }

    if (sourceUri.startsWith("data:")) {
      return res.json({ url: sourceUri });
    }

    const sourcePath = sourceUri.replace("file://", "");
    if (!fs.existsSync(sourcePath)) {
      return res.status(404).json({ error: "Source media file not found" });
    }

    const ext = path.extname(filename || sourcePath) || ".bin";
    const safeBase = path
      .basename(filename || `case-${caseId}`, ext)
      .replace(/[^a-zA-Z0-9-_]+/g, "-")
      .slice(0, 80) || `case-${caseId}`;

    const uploadDir = path.resolve(process.cwd(), "uploads", "case-media");
    fs.mkdirSync(uploadDir, { recursive: true });

    const storedName = `${Date.now()}-${caseId}-${safeBase}${ext}`;
    const destinationPath = path.join(uploadDir, storedName);
    fs.copyFileSync(sourcePath, destinationPath);

    const forwardedHost = req.header("x-forwarded-host");
    const host = forwardedHost || req.get("host");
    const forwardedProto = req.header("x-forwarded-proto");
    const protocol = forwardedProto || req.protocol || "https";

    return res.json({
      url: `${protocol}://${host}/uploads/case-media/${storedName}`,
      contentType: contentType || null,
    });
  } catch (error: any) {
    console.error("Share media error:", error?.message || error);
    res.status(500).json({ error: "Failed to share media" });
  }
});
