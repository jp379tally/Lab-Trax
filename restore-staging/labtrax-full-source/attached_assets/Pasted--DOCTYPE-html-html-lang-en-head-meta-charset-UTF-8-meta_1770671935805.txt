    // --- UI Controllers ---

    function setRole(role) {
        state.role = role;
        
        // Update Buttons
        const btns = { tech: document.getElementById('role-tech'), admin: document.getElementById('role-admin') };
        Object.keys(btns).forEach(key => {
            btns[key].className = key === role 
                ? "px-6 py-2 rounded-lg text-sm font-bold bg-white shadow-sm transition-all"
                : "px-6 py-2 rounded-lg text-sm font-bold text-slate-500 hover:text-slate-800 transition-all";
        });

        // Update Explainer
        const badge = document.getElementById('logic-badge');
        badge.textContent = role === 'tech' ? 'Technician Logic' : 'Administrator Logic';
        badge.className = role === 'tech' 
            ? "bg-blue-100 text-blue-700 text-xs font-black px-3 py-1 rounded-full uppercase tracking-widest inline-block mb-4"
            : "bg-indigo-100 text-indigo-700 text-xs font-black px-3 py-1 rounded-full uppercase tracking-widest inline-block mb-4";
            
        document.getElementById('logic-title').textContent = reqs[role].title;
        document.getElementById('logic-desc').textContent = reqs[role].desc;
        
        const specContainer = document.getElementById('logic-specs');
        specContainer.innerHTML = reqs[role].specs.map(s => `
            <div class="flex items-center gap-4 p-4 rounded-2xl border border-slate-100">
                <div class="w-10 h-10 rounded-full flex items-center justify-center ${role === 'tech' ? 'bg-blue-50 text-blue-600' : 'bg-indigo-50 text-indigo-600'}"><i class="fa-solid ${s.icon}"></i></div>
                <div>
                    <h5 class="font-bold text-sm">${s.title}</h5>
                    <p class="text-xs text-slate-400">${s.detail}</p>
                </div>
            </div>
        `).join('');

        // Reset Mockup
        if (role === 'admin') {
            loadScreen('lock');
        } else {
            loadScreen('dashboard');
        }
    }

    }
