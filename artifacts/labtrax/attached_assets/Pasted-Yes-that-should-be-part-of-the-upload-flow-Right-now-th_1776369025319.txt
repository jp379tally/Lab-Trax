Yes — that should be part of the upload flow.

Right now the drag-and-drop zone is acting like a dumb thumbnail tray. It needs to become an **attachment review step**:

1. user uploads or screenshots a file from iPhone
2. file lands in the drop zone
3. app immediately prompts for a note
4. tapping the thumbnail opens a preview modal
5. preview modal lets them edit the note, not just delete

## What to change

### 1. Store attachments as objects, not plain URIs

Do **not** keep this as:

```ts
string[]
```

Use:

```ts
type UploadAttachment = {
  id: string;
  uri: string;
  type: "image" | "video" | "pdf" | "file";
  name?: string;
  note: string;
  createdAt: number;
};
```

State:

```ts
const [attachments, setAttachments] = useState<UploadAttachment[]>([]);
const [selectedAttachment, setSelectedAttachment] = useState<UploadAttachment | null>(null);
const [noteModalVisible, setNoteModalVisible] = useState(false);
const [previewVisible, setPreviewVisible] = useState(false);
const [draftNote, setDraftNote] = useState("");
```

---

### 2. Prompt for notes immediately after upload

When a screenshot, photo, video, or file is added, create the attachment first, then immediately open the note prompt.

```ts
function detectAttachmentType(asset: any): UploadAttachment["type"] {
  if (asset?.mimeType?.startsWith("image/")) return "image";
  if (asset?.mimeType?.startsWith("video/")) return "video";
  if (asset?.mimeType === "application/pdf") return "pdf";
  return "file";
}

function createAttachment(asset: any): UploadAttachment {
  return {
    id: `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
    uri: asset.uri,
    type: detectAttachmentType(asset),
    name: asset.name || asset.fileName || undefined,
    note: "",
    createdAt: Date.now(),
  };
}

function addAttachmentAndPrompt(asset: any) {
  const attachment = createAttachment(asset);

  setAttachments((prev) => [...prev, attachment]);
  setSelectedAttachment(attachment);
  setDraftNote("");
  setNoteModalVisible(true);
}
```

Then in every upload path:

* screenshot upload
* photo picker
* file explorer
* video picker

replace direct pushes with:

```ts
addAttachmentAndPrompt(asset);
```

### 3. Support iPhone screenshots directly

As long as the screenshot is in Photos or Files, the user should be able to upload it normally from iPhone.

For document picker:

```ts
const result = await DocumentPicker.getDocumentAsync({
  type: ["image/*", "video/*", "application/pdf", "*/*"],
  copyToCacheDirectory: true,
  multiple: false,
});

if (!result.canceled && result.assets?.length) {
  for (const asset of result.assets) {
    addAttachmentAndPrompt(asset);
  }
}
```

For image picker:

```ts
const result = await ImagePicker.launchImageLibraryAsync({
  mediaTypes: ["images", "videos"],
  quality: 1,
});

if (!result.canceled && result.assets?.length) {
  for (const asset of result.assets) {
    addAttachmentAndPrompt(asset);
  }
}
```

That covers screenshots saved on iPhone.

### 4. Make thumbnails open a preview + notes editor

Instead of only delete-on-tap, tapping should open the selected attachment.

```tsx
<Pressable
  onPress={() => {
    setSelectedAttachment(item);
    setDraftNote(item.note || "");
    setPreviewVisible(true);
  }}
  style={styles.attachmentThumbWrap}
>
  {item.type === "image" ? (
    <Image source={{ uri: item.uri }} style={styles.attachmentThumb} contentFit="cover" />
  ) : item.type === "video" ? (
    <View style={styles.attachmentPlaceholder}>
      <Ionicons name="play-circle" size={30} color="#1e293b" />
      <Text style={styles.attachmentLabel}>Video</Text>
    </View>
  ) : item.type === "pdf" ? (
    <View style={styles.attachmentPlaceholder}>
      <Ionicons name="document-text-outline" size={28} color="#1e293b" />
      <Text style={styles.attachmentLabel}>PDF</Text>
    </View>
  ) : (
    <View style={styles.attachmentPlaceholder}>
      <Ionicons name="document-outline" size={28} color="#1e293b" />
      <Text style={styles.attachmentLabel}>File</Text>
    </View>
  )}
</Pressable>
```

### 5. Add note modal right after upload

```tsx
<Modal visible={noteModalVisible} transparent animationType="slide">
  <View style={styles.modalBackdrop}>
    <View style={styles.modalCard}>
      <Text style={styles.modalTitle}>Add a note</Text>
      <Text style={styles.modalSubtitle}>
        Describe what this file is before adding it to the case.
      </Text>

      <TextInput
        value={draftNote}
        onChangeText={setDraftNote}
        placeholder="Example: Bite open, prep photo, patient screenshot, shade reference..."
        multiline
        style={styles.noteInput}
      />

      <View style={styles.modalActions}>
        <Pressable
          onPress={() => {
            setNoteModalVisible(false);
            setSelectedAttachment(null);
            setDraftNote("");
          }}
          style={styles.secondaryBtn}
        >
          <Text>Skip</Text>
        </Pressable>

        <Pressable
          onPress={() => {
            if (!selectedAttachment) return;

            setAttachments((prev) =>
              prev.map((a) =>
                a.id === selectedAttachment.id
                  ? { ...a, note: draftNote.trim() }
                  : a
              )
            );

            setNoteModalVisible(false);
            setSelectedAttachment(null);
            setDraftNote("");
          }}
          style={styles.primaryBtn}
        >
          <Text style={{ color: "#fff" }}>Save Note</Text>
        </Pressable>
      </View>
    </View>
  </View>
</Modal>
```

### 6. Add preview modal with editable notes

```tsx
<Modal visible={previewVisible} animationType="slide">
  <View style={styles.previewScreen}>
    <View style={styles.previewHeader}>
      <Pressable onPress={() => setPreviewVisible(false)}>
        <Ionicons name="close" size={28} color="#0f172a" />
      </Pressable>
      <Text style={styles.previewTitle}>Attachment</Text>
      <Pressable
        onPress={() => {
          if (!selectedAttachment) return;
          setAttachments((prev) => prev.filter((a) => a.id !== selectedAttachment.id));
          setPreviewVisible(false);
          setSelectedAttachment(null);
        }}
      >
        <Ionicons name="trash-outline" size={24} color="#dc2626" />
      </Pressable>
    </View>

    {selectedAttachment?.type === "image" ? (
      <Image
        source={{ uri: selectedAttachment.uri }}
        style={styles.previewImage}
        contentFit="contain"
      />
    ) : (
      <View style={styles.previewFallback}>
        <Ionicons
          name={
            selectedAttachment?.type === "video"
              ? "play-circle-outline"
              : "document-text-outline"
          }
          size={52}
          color="#334155"
        />
        <Text style={styles.previewFallbackText}>
          {selectedAttachment?.name || selectedAttachment?.type}
        </Text>
      </View>
    )}

    <Text style={styles.noteLabel}>Notes</Text>
    <TextInput
      value={draftNote}
      onChangeText={setDraftNote}
      placeholder="Add notes for this attachment"
      multiline
      style={styles.noteInput}
    />

    <Pressable
      onPress={() => {
        if (!selectedAttachment) return;
        setAttachments((prev) =>
          prev.map((a) =>
            a.id === selectedAttachment.id ? { ...a, note: draftNote.trim() } : a
          )
        );
        setPreviewVisible(false);
      }}
      style={styles.primaryBtn}
    >
      <Text style={{ color: "#fff" }}>Save Changes</Text>
    </Pressable>
  </View>
</Modal>
```

## Resulting behavior

After this change, the user flow becomes:

* take screenshot on iPhone
* upload directly into LabTrax
* file appears in drop zone
* note prompt opens immediately
* user saves note
* later they can tap the thumbnail
* preview opens
* they can edit note or delete attachment

That is the behavior you want.

## Best UX rule

Make the note prompt **optional but immediate**:

* do not force typing before upload finishes
* do prompt right after upload
* allow “Skip”
* keep notes editable later

That keeps the workflow fast without losing context.
